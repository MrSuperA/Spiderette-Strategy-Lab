// ═══ 网络层（超时/重试/去重/自适应） ═══
const NET={timeout:5000,retries:3,retryDelay:500,inflight:new Map()};

async function api(p,opts={}){
  // 请求去重：同一路径正在请求中则复用
  const key=p+(opts.method||'GET');
  if(NET.inflight.has(key))return NET.inflight.get(key);

  const promise=_fetchWithRetry(p,opts);
  NET.inflight.set(key,promise);
  promise.finally(()=>NET.inflight.delete(key));
  return promise;
}

async function _fetchWithRetry(p,opts,retry=0){
  const ctrl=new AbortController();
  const timer=setTimeout(()=>ctrl.abort(),NET.timeout);
  try{
    const r=await fetch('/api/'+p,{headers:{'Content-Type':'application/json'},signal:ctrl.signal,...opts});
    clearTimeout(timer);
    if(!r.ok)throw new Error(`HTTP ${r.status}`);
    return await r.json();
  }catch(e){
    clearTimeout(timer);
    if(retry<NET.retries&&e.name!=='AbortError'){
      await new Promise(r=>setTimeout(r,NET.retryDelay*(retry+1)));
      return _fetchWithRetry(p,opts,retry+1);
    }
    if(e.name==='AbortError')throw new Error('请求超时');
    throw e;
  }
}

async function newGame(){
  if(state==='running'||state==='paused')return;
  const d=await api('new-game',{method:'POST',body:JSON.stringify({seed:Math.floor(Math.random()*1e5),difficulty:+$i('selDiff').value,strategy:$i('selStrat').value})});

// ── 牌桌渲染（增量更新） ──
let _prevCols=null; // 上一次的列数据快照

function renderBoard(s){
  const cols=s.columns||[];
  const grid=$i('boardGrid');

  // 首次渲染：创建全部 DOM
  if(!grid.children.length||!_prevCols||_prevCols.length!==cols.length){
    let h='';
    for(let i=0;i<cols.length;i++){
      h+=`<div class="col" id="col${i}"><div class="col-hd">${i+1}</div><div class="col-body" id="colb${i}"></div></div>`;
    }
    grid.innerHTML=h;
    _prevCols=null;
  }

  // 增量更新：只更新变化的列
  for(let i=0;i<cols.length;i++){
    const c=cols[i];
    const prev=_prevCols?_prevCols[i]:null;
    const fd=c.face_down||0;
    const fu=c.face_up||[];

    // 检查是否变化
    const pfd=prev?(prev.face_down||0):-1;
    const pfu=prev?(prev.face_up||[]):null;
    const fuKey=fu.map(k=>k.rank+k.suit).join(',');
    const pfuKey=pfu?pfu.map(k=>k.rank+k.suit).join(','):'';

    if(fd===pfd&&fuKey===pfuKey)continue; // 未变化，跳过

    // 重建这一列的 DOM
    const body=$i('colb'+i);
    if(!body)continue;
    let h='';
    if(!fd&&!fu.length){
      h='<div class="col-empty">空</div>';
    }else{
      for(let j=0;j<fd;j++)h+='<div class="card card-fd"></div>';
      for(let j=0;j<fu.length;j++){
        const k=fu[j];
        h+=`<div class="card card-fu ${k.red?'red':'black'} ${j===fu.length-1?'card-top':''}">${k.rank}${k.suit}</div>`;
      }
    }
    body.innerHTML=h;
  }

  _prevCols=cols;
}

  else if(m.action==='complete')desc='完成序列';

  else desc=m.action||'移牌';

  const elapsed=step.elapsed_ms?step.elapsed_ms.toFixed(1)+'ms':'-';
  const legal=step.legal_moves||0;
  const msg=`#${step.step} ${desc}  耗时${elapsed}  合法${legal}`;

// ── 快照（含移动描述） ──
function appendSnap(step){
  if(!step||!step.state)return;
  const cols=step.state.columns||[];
  const m=step.move;
  let desc='开始';
  if(m){
    if(m.action==='deal')desc='发牌';
    else if(m.action==='complete')desc='完成序列';
    else if(m.action==='move')desc=`列${(m.src||0)+1}→列${(m.dst||0)+1}`;
    else desc=m.action||'移牌';
  }

  let txt=`步数 #${step.step}  ${desc}\n`;
  for(let i=0;i<cols.length;i++){
    const c=cols[i],fd=c.face_down||0,fu=c.face_up||[];
    let ln=`列${String(i+1).padStart(2)}: `.padEnd(7);
    for(let j=0;j<fd;j++)ln+='?? ';
    fu.forEach(k=>ln+=k.rank+k.suit+' ');
    txt+=ln+'\n';
  }
  txt+=`── 完成:${step.state.completed||0}  发牌:${step.state.stock_remaining||0}  步数:${step.state.move_count||0} ──`;

  snapHistory.push(txt);
  if(snapHistory.length>SNAP_MAX)snapHistory.shift();
  snapIdx=snapHistory.length-1;
  if(snapAuto){$i('snapText').textContent=txt;$i('snapCtr').textContent=snapHistory.length+'/'+snapHistory.length}
}
function snapNav(dir){
  if(!snapHistory.length)return;snapAuto=false;
  if(dir==='first')snapIdx=0;else if(dir==='prev')snapIdx=Math.max(0,snapIdx-1);
  else if(dir==='next')snapIdx=Math.min(snapHistory.length-1,snapIdx+1);else if(dir==='last'){snapIdx=snapHistory.length-1;snapAuto=true}
  $i('snapText').textContent=snapHistory[snapIdx];$i('snapCtr').textContent=(snapIdx+1)+'/'+snapHistory.length;
}

function connectSSE(){
  const es=new EventSource('/api/stream');
  es.onopen=function(){_sseOk=true;_sseRetry=0};
  es.onmessage=function(e){
    try{
      const d=JSON.parse(e.data);

        _pendingState=d.state;_stepBuffer.push(d.step);

        totalMoves++;stepTimes.push(d.step?.elapsed_ms||0);legalCounts.push(d.step?.legal_moves||0);
        _factorDirty=true;
        // SSE sequence tracking

          _sseLastSeq=d._sse_seq;_sseSeq=d._sse_seq;

          const dot=$i('syncDot');const txt=$i('syncText');

          if(txt)txt.textContent='SEQ:'+_sseSeq+' 丢包:'+_sseDropCount;
        }
        // Update strategy stats panel
        if(d.strategy_stats){
          const ss=d.strategy_stats;
          const box=$i('stratStatsBox');if(box)box.style.display='';
          const sel=$i('selStrat');
          const sn=$i('dkStratName');if(sn)sn.textContent=sel.options[sel.selectedIndex].text;
          const si=$i('dkStatIter');if(si)si.textContent=ss.iterations||'\u2014';
          const st=$i('dkStatTree');if(st)st.textContent=ss.tree_size||'\u2014';
          const sm=$i('dkStatMem');if(sm)sm.textContent=ss.memory_hit_rate!==undefined?(ss.memory_hit_rate*100).toFixed(1)+'%':'\u2014';
          const sl=$i('dkStatLegal');if(sl)sl.textContent=d.step?.legal_moves||'\u2014';
          const sq=$i('dkStatSeq');if(sq)sq.textContent=_sseSeq;
          const sd=$i('dkStatDrop');if(sd){sd.textContent=_sseDropCount;sd.style.color=_sseDropCount>0?'var(--red)':'var(--green)';}
        }
      }else if(d.type==='auto_done'){
        if(state==='stopped')return;
        if(d.result)lastResult=d.result;
        if(d.state){_pendingState=d.state;}
        flushRender();
        state='ended';updateButtons();
        const ssb=$i('stratStatsBox');if(ssb)ssb.style.display='none';
        const si2=$i('syncIndicator');if(si2)si2.style.display='none';
        onGameEnd(d.state||{move_count:totalMoves,completed:lastResult?.completed||0});
        toast(lastResult?.outcome==='win'?'\ud83c\udf89 \u80dc\u5229\uff01':'\ud83d\udc80 \u6b7b\u5c40',lastResult?.outcome==='win'?'ok':'err');
      }else if(d.type==='auto_paused'){
        if(state==='stopped')return;
        if(d.state){_pendingState=d.state;flushRender();}
        const ssb2=$i('stratStatsBox');if(ssb2)ssb2.style.display='none';
        const si3=$i('syncIndicator');if(si3)si3.style.display='none';
      }else if(d.type==='new_game'&&d.state){
        _prevCols=null;_pendingState=d.state;
        _sseSeq=0;_sseLastSeq=0;_sseDropCount=0;
          flushRender();
      }
      _sseOk=true;
    }catch(ex){}
  };
  es.onerror=function(){
    _sseOk=false;es.close();
    _sseRetry++;setTimeout(connectSSE,Math.min(1000*Math.pow(2,_sseRetry),30000));
  };
}
connectSSE();

function flushRender(){
  if(!_pendingState)return;
  const s=_pendingState;_pendingState=null;

  // 处理缓冲的步骤：每步都记录日志和快照
  if(_stepBuffer.length>0){
    _stepBuffer.forEach(step=>{
      appendSnap(step);
      logStep(step);
    });
    prevState=s;
    _stepBuffer=[];
  }

  // 更新策略监控
  if(_pendingStrategyStats){
    const ss=_pendingStrategyStats;
    if(ss.iterations!==undefined)$i('siIter').textContent=ss.iterations;
    if(ss.tree_size!==undefined)$i('siSim').textContent=ss.tree_size;
    if(ss.memory_hit_rate!==undefined)$i('siExplore').textContent=(ss.memory_hit_rate*100).toFixed(0)+'%';
    _pendingStrategyStats=null;
  }

  updateUI(s);
}

  if(_factorDirty&&now-_lastFactorCalc>FACTOR_MS&&state==='running'){

    try{
      const strat=$i('selStrat').value;const diff=+$i('selDiff').value;

      if(s.gpu_name){smoothGpu=smoothGpu===0?s.gpu_load:smoothGpu*0.7+s.gpu_load*0.3;const gpu=Math.round(smoothGpu*10)/10;$i('perfGpu').textContent=gpu.toFixed(1)+'%';$i('perfGpuBar').style.width=gpu+'%';$i('perfGpuName').textContent=s.gpu_name;$i('perfGpuTemp').textContent=s.gpu_temp?s.gpu_temp+'°C':''}
      else{$i('perfGpu').textContent='未检测到';$i('perfGpuBar').style.width='0%';$i('perfGpuName').textContent='无独立GPU';$i('perfGpuTemp').textContent=''}
    }catch(e){}
  }
}
